356:     /**
357:      * extract and output keywords from text.
358:      * Kewords are sorted by relevance (descending)
359:      * Words and punctuations exclusions are located in language file.
360:      *
361:      * @access public
362:      * @static
363:      * @param mixed $str
364:      * @param array $args. (default: array()):
365:      * "min_occur" (int): Minimum of word occurrences in $str to consider,
366:      * "min_length" (int): Minimum words length to consider,
367:      * "max" (int): Maximum of keywords to consider,
368:      * "separator" (string): Ouput keywords separator
369:      * @return String
370:      */
371:     public static function keywords($str, $args = array())
372:     {
373:         self::extend($args, __FUNCTION__);
374: 
375:         $commons        = self::get_lang('text', 'keywords');
376:         $replacements   = array('', '', ' ');
377:         $patterns           = array(
378:             '/\b\w{0,'.($args['min_length'] - 1).'}\b/', // less than $args['min_length']
379:             "#(".implode(' | ', $commons['words']).")#i", // ignored common words
380:             '/\s\s+/' // extra whitespaces
381:         );
382:         // replace punctuations with whitespaces and apply patterns
383:         $str = preg_replace($patterns, $replacements, str_replace($commons['punct'], ' ', $str));
384:         // prepare keywords if $str not empty
385:         if ($str)
386:         {
387:             // trim and lower case str, create an array and add occurences count as values
388:             $words = array_count_values(explode(" ", strtolower(trim($str))));
389:             // sort the words highest count > lowest count.
390:             arsort($words);
391:             // slice array to feet max keywords count
392:             $words = array_slice($words, 0, $args['max'], true);
393:             // filter words of < 'min_occur' length
394:             $words = array_filter($words, function($count) use($args){
395:                     return ($count < $args['min_occur']) ? false : true;
396:                 });
397:             // return imploded keywords array, according to 'separator' pattern
398:             $str = implode($args['separator'], array_keys($words));
399:         }
400:         return $str;
401:     }
